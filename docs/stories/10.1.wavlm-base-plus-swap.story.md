## Dev Notes

### Modelo recomendado
- **`microsoft/wavlm-base-plus`** — 94M params, ~400MB, eq ou superior wav2vec2-base-superb-er em paralinguística.
- **NÃO usar:** `microsoft/wavlm-large` (overkill, ~600MB sem ganho proporcional), checkpoints fine-tuned ASR.

### Layer recipe (de research 04-deep-read)
- Layers 3-10 (de 12 total no base+) cobrem prosódia/voice quality.
- Weighted-sum learnable: `weights = softmax(nn.Parameter(torch.ones(8)))`.
- Pool temporal: mean-pooling sobre tempo após weighted-sum.

### Esqueleto código sugerido pra `_wavlm_emotion.py`

```python
"""WavLM-base+ Emotion (Story 10.1).

Substitui wav2vec2-base-superb-er. Layers 3-10 weighted-sum, NÃO ASR-tuned.
"""
from __future__ import annotations
import os
from typing import Any
import structlog

logger = structlog.get_logger()

MODEL_ID = "microsoft/wavlm-base-plus"
LAYER_START = 3
LAYER_END = 10  # inclusive

def load_wavlm_emotion() -> dict[str, Any] | None:
    try:
        from transformers import AutoFeatureExtractor, AutoModel
        import torch
        import torch.nn as nn
    except ImportError:
        logger.warning("transformers_not_installed_for_wavlm")
        return None
    try:
        model = AutoModel.from_pretrained(MODEL_ID, output_hidden_states=True)
        # Reject ASR-tuned: AutoModel não tem .config.task = "ctc",
        # mas verificar via class name conservadoramente
        if "CTC" in type(model).__name__:
            raise RuntimeError(f"Refusing ASR-tuned: {type(model).__name__}")
        feature_extractor = AutoFeatureExtractor.from_pretrained(MODEL_ID)
        model.eval()

        # Weighted-sum learnable sobre layers 3-10 (8 layers)
        n_layers = LAYER_END - LAYER_START + 1
        layer_weights = nn.Parameter(torch.ones(n_layers))

        return {
            "model": model,
            "feature_extractor": feature_extractor,
            "layer_weights": layer_weights,
        }
    except Exception as e:
        logger.warning("wavlm_load_failed", error=str(e))
        return None

def extract_features(bundle: dict, audio_path: str):
    import torch
    import librosa
    audio, sr = librosa.load(audio_path, sr=16000)
    inputs = bundle["feature_extractor"](
        audio, sampling_rate=16000, return_tensors="pt"
    )
    with torch.no_grad():
        outputs = bundle["model"](**inputs, output_hidden_states=True)
    # outputs.hidden_states tuple of 13 tensors (embedding + 12 layers)
    # layers indexados a partir de 1 (layer 0 = embedding)
    selected = torch.stack(outputs.hidden_states[LAYER_START:LAYER_END+1])  # (8, B, T, 768)
    weights = torch.softmax(bundle["layer_weights"], dim=0)
    weighted = (selected * weights.view(-1, 1, 1, 1)).sum(dim=0)  # (B, T, 768)
    pooled = weighted.mean(dim=1)  # (B, 768)
    if torch.isnan(pooled).any() or torch.isinf(pooled).any():
        return None
    return pooled.squeeze(0)  # (768,)

def predict_vad(embedding) -> dict[str, float]:
    """Método B (fallback) — mapping via emotion classification + EMOTION_TO_VAD.

    Método A (Linear+Sigmoid) requer training data — fica pra Epic 11+.
    """
    # ... (escolha @dev no Pre-Flight)
    raise NotImplementedError("Choose method A or B in Task 1.3")

def infer_emotions(bundle: dict, audio_path: str) -> dict | None:
    """Interface compatível com _emotion_ml.infer_emotions (facilita swap)."""
    embedding = extract_features(bundle, audio_path)
    if embedding is None:
        return None
    vad = predict_vad(embedding)
    return {
        "vad_ml": vad,
        "model_id": MODEL_ID,
        "extraction_method": "wavlm_layers_3_10_weighted_sum",
    }
```
